function dcsA(N,V){
	return "&"+N+"="+dcsEscape(V, RE);
}

function dcsEscape(S, REL){
	if (typeof(REL)!="undefined"){
		var retStr = new String(S);
		for (R in REL){
			retStr = retStr.replace(REL[R],R);
		}
		return retStr;
	}
	else{
		return escape(S);
	}
}

function dcsEncode(S){
	return (typeof(encodeURIComponent)=="function")?encodeURIComponent(S):escape(S);
}

function dcsMeta(){
	var elems;
	if (document.all){
		elems=document.all.tags("meta");
	}
	else if (document.documentElement){

	}
	if (typeof(elems)!="undefined"){
		for (var i=1;i<=elems.length;i++){
			var meta=elems.item(i-1);
			if (meta.name){
				if (meta.name.indexOf('WT.')==0){
					WT[meta.name.substring(3)]=(gI18n&&(meta.name.indexOf('WT.ti')==0))?dcsEscape(dcsEncode(meta.content),I18NRE):meta.content;

function dcsJV(){
	var agt=navigator.userAgent.toLowerCase();

	var nn=((agt.indexOf("mozilla")!=-1)&&(agt.indexOf("compatible")==-1));
	var nn4=(nn&&(major==4));
	var nn6up=(nn&&(major>=5));
	var ie=((agt.indexOf("msie")!=-1)&&(agt.indexOf("opera")==-1));
	var ie4=(ie&&(major==4)&&(agt.indexOf("msie 4")!=-1));
	var ie5up=(ie&&!ie4);
	var op=(agt.indexOf("opera")!=-1);
	var op5=(agt.indexOf("opera 5")!=-1||agt.indexOf("opera/5")!=-1);
	var op6=(agt.indexOf("opera 6")!=-1||agt.indexOf("opera/6")!=-1);
	var op7up=(op&&!op5&&!op6);
	var jv="1.1";
	if (nn6up||op7up){
		jv="1.5";
	}
	else if ((mac&&ie5up)||op6){
		jv="1.4";
	}
	else if (ie5up||nn4||op5){
		jv="1.3";
	}
	else if (ie4){

function getQueryParameter(url,param)
{
  var index = url.indexOf(param+"=");
  var value;
  
  if (index != -1)
  {
    var secondIndex = index + param.length + 1;
    var len = url.substring(secondIndex).indexOf("&");
    if (len == -1)
      value = url.substring(secondIndex);
    else
      value = url.substring(secondIndex, secondIndex + len)
  }
  else
    value = null;
    
  return value;
}

	else if(urlPcs[1] == "statsweekly")
	    WT.cg_s = "Stats Weekly";
	else if(url.match("newsletter","i"))
	    WT.cg_s = "Newsletter";
}

// set content groups for the bet guide pages
function getBetGuideGroup(url, urlPcs)
{
	WT.cg_n = "Bet Guide";
	
	if(url.match("/articles/(archive|articles)\.aspx","i"))
	    WT.cg_s = "101";
	else if(url.match("sportsbetting/betting-faq\.aspx","i"))
	    WT.cg_s = "FAQ";
	else if(url.match("sportsbetting/glossary\.aspx","i"))
	    WT.cg_s = "Glossary";
}

// set content groups for the other pages
function getOtherGroup(url)
{
	WT.cg_n = "Other";
	
	if(url.match("advertise-on-covers\.aspx","i"))
	    WT.cg_s = "Advertise";
	else if(url.match("sitemap\.aspx","i"))
	    WT.cg_s = "Site Map";
	else if(url.match("help\.aspx","i"))
	    WT.cg_s = "Help Pages";
	else if(url.match("about\.aspx","i"))
	    WT.cg_s = "About";
	else if(url.match("contact\.aspx","i"))
	    WT.cg_s = "Contact Pages";
	else if(url.match("privacy\.aspx","i"))
	    WT.cg_s = "Privacy";
	else if(url.match("terms\.aspx","i"))
	    WT.cg_s = "Terms";
	else
	    WT.cg_s = "Other";
}

// wrapper function to call the other functions at the appropriate times
function setContentGroups()
{
	var url = window.location.pathname;
	var qry = window.location.search;
	var host = window.location.hostname;
	var urlPcs = url.split("/");
	
	if((urlPcs[1] == "index.aspx") && (host.match("spaces","i") == null ))
	    WT.cg_n = "Index Page";
    else if((urlPcs[1] == "articles") && (url.match("/archive\.aspx","i") == null) && qry.match("tid=70","i") == null)
        getArticlesGroup(url);
    else if(urlPcs[1] == "postingforum")
        getForumGroup(url,qry);
    else if(((urlPcs[1] == "sports") && (url.match("myOdds\.aspx","i") == null)) || (url.match("scores-matchups","i")) || (qry.match("data","i")))
        getSportsGroup(url, qry, urlPcs);
    else if((urlPcs[1] == "custom-scores.aspx") || (url.match("myOdds\.aspx","i")))
        getCustomGroup(url, urlPcs);
    else if(host.match("spaces","i"))
        getSpacesGroup(url);
    else if((urlPcs[1].match("(mobile|rss|odds_ex|statsweekly)","i")) || (url.match("newsletter","i")))
        getFeaturesGroup(url, urlPcs);
    else if(url.match("(/articles/(archive|articles)\.aspx|sportsbetting/(betting-faq|glossary)\.aspx)","i"))
        getBetGuideGroup(url);
    else
        getOtherGroup(url);
}
